## <a name="example"></a><span data-ttu-id="89185-107">Esempio</span><span class="sxs-lookup"><span data-stu-id="89185-107">Example</span></span>

{  
   public virtual void Finalize() {}   // CS0465

{  
   ~B() {}  
}
